    \begin{align*} c^2 \amp = a^2+b^2 \amp \amp \blert{\text{Substitute }a=8~ \text{and }b=6~ \text{and evaluate.}}\\ c^2 \amp = 8^2+6^2=64+36=100 \amp \amp \blert{\text{Take square roots.}}\\ c \amp = \sqrt{100} = 10 \end{align*}
