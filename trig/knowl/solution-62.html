\begin{align*} x \amp = r \cos 292\degree \amp\amp \text{and} \amp y \amp = r \sin 292\degree\\ \amp = 6(0.3746) \amp\amp\amp \amp = 6(-0.9272)\\ \amp = 2.2476 \amp\amp\amp \amp = -5.5632 \end{align*}
