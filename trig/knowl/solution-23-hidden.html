\begin{equation*} \sin 50\degree = \dfrac {\text{opposite}}{\text{hypotenuse}} \end{equation*}

\begin{equation*} 0.7660 = \dfrac {x}{18} \end{equation*}

\begin{equation*} x = 18(0.7660)=13.788 \end{equation*}
