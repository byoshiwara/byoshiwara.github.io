\begin{equation*} \begin{aligned}[t] \dfrac{\dfrac{\sin \theta}{\cos \theta} - \sin \theta \cos \theta}{\sin \theta \cdot \dfrac{\sin \theta}{\cos \theta}} \amp = \dfrac{\sin \theta - \sin \theta \cos^2 \theta}{\sin^2 \theta}\\ \amp = \dfrac{\sin \theta (1 - \cos^2 \theta)}{\sin^2 \theta} = \dfrac{\sin \theta \sin^2 \theta}{\sin^2 \theta} = \sin \theta\\ \end{aligned} \end{equation*}
