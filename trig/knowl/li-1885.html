\begin{align*} 2 \tan (\alert{135\degree}) + 3\sin (\alert{30\degree}) \amp = 2(-1) + 3(\dfrac{1}{2})\\ \amp= -2 + \dfrac{3}{2} =\dfrac{-1}{2} \end{align*}
