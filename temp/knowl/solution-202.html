\begin{equation*} \begin{aligned}[t] 3 \amp =3\sin 2\theta \amp\amp \blert{\text{Divide both sides by } 3.}\\ 1 \amp = \sin 2\theta \amp\amp \blert{\text{Apply the arcsine function to both sides.}}\\ \dfrac{\pi}{2} \amp = 2\theta \amp\amp \blert{\text{Divide both sides by } 2.}\\ \dfrac{\pi}{4} \amp = \theta\\ \end{aligned} \end{equation*}
