\begin{equation*} P(h) = 30 (10)^{-0.09h} \end{equation*}

\begin{align*} {\text{Xmin}} \amp = 0 \amp\amp {\text{Xmax}} = 9.4\\ {\text{Ymin}} \amp = 0 \amp\amp {\text{Ymax}} = 30 \end{align*}
