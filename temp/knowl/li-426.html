\begin{equation*} \tan 58\degree = \dfrac{\text{opposite}}{\text{adjacent}} \end{equation*}

\begin{equation*} 1.6003 = \frac{h}{300} \end{equation*}

\begin{equation*} h = 300(1,6003) = 480.0 \end{equation*}
