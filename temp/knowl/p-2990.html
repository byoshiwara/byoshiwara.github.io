\begin{align*} 6w-6 \amp = 36 \amp\amp \blert{\text{Add 6 to both sides.}}\\ 6w \amp = 42 \amp\amp \blert{\text{Divide both sides by 6.}}\\ w \amp = 7 \end{align*}
