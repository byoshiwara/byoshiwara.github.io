\begin{equation*} \begin{aligned}[t] c \amp = \dfrac{15.497 \pm \sqrt{(-15.497)^{2} - 4(1)(55)}}{2(1)} \amp\amp \blert{\text{Simplify.}}\\ \amp = \dfrac{15.497 \pm 4.490}{2} = 5.503~ \text{or}~ 9.994\\ \end{aligned} \end{equation*}
