\begin{equation*} \begin{aligned}[t] \cos 35 \degree \amp = \dfrac {\text{adjacent}}{\text{hypotenuse}} \\ \dfrac{\sqrt{3}}{2} \amp = \dfrac{h}{8} \\ h \amp = 8(\dfrac{\sqrt{3}}{2}) = 4\sqrt{3} \\ \end{aligned} \end{equation*}
