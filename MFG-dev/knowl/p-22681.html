\begin{align*} x^2+4x^2\amp =225\amp\amp\blert{\text{Combine like terms.}} \\ 5x^2\amp = 225\amp\amp\blert{\text{Divide both sides by 5.}} \\ x^2\amp = 45 \end{align*}
