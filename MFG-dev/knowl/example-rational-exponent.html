  1. \begin{align*} 81^{3/4} \amp = \left(81^{1/4}\right)^3 \\ \amp = 3^3 = 27 \end{align*}

  2. \begin{align*} -27^{5/3} \amp = -\left(27^{1/3}\right)^5 \\ \amp = -3^5 = -243 \end{align*}

  3. \begin{align*} 27^{-2/3} \amp = \frac{1}{\left(27^{1/3}\right)^2} \\ \amp = \frac{1}{3^2}= \frac{1}{9} \end{align*}

  4. \begin{align*} 5^{3/2} \amp = \left(5^{1/2}\right)^3 \\ \amp \approx (2.236)^3 \approx 11.180 \end{align*}
