    \begin{equation*} \begin{aligned}[t] \text{when } w \amp= 1, \amp\amp L = 10d^2 \\ \text{when } w \amp= 2, \amp\amp L = 20d^2 \\ \text{when } w \amp= 3, \amp\amp L = 30d^2 \\ \text{when } w \amp= 4, \amp\amp L = 40d^2 \end{aligned} \end{equation*}
